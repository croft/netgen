main
  [template_instantiation*] {

  lb.wb.CreateTemplate(
    json_spec_file = "workbookutil/template.json",
    app_prefix = "workbookutil"
  )
  ;
  ($template_instantiation = template_instantiation) <~ lb.wb.util.GetTemplateInstantiationRefByTemplate(
    app_prefix = "workbookutil",
    template_name = "region"
  )
  ;
  forall<max=3>(ti in $template_instantiation) {
    lb.wb.util.CreateWorkbook(
      app_prefix = "workbookutil",
      template_instantiation = $ti
    )
  }
}
